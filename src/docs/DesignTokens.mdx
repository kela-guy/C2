## Token Usage

```ts
import { CARD_TOKENS } from "@/primitives";

// Container styles
style={{
  backgroundColor: CARD_TOKENS.container.bgColor,    // #1A1A1A
  borderColor: CARD_TOKENS.container.borderColor,     // #333333
  borderRadius: CARD_TOKENS.container.borderRadius,   // 8px
}}

// Header
style={{
  padding: `${CARD_TOKENS.header.paddingY}px ${CARD_TOKENS.header.paddingX}px`,
}}

// Icon box
style={{
  width: CARD_TOKENS.iconBox.size,      // 32px
  height: CARD_TOKENS.iconBox.size,
  borderRadius: CARD_TOKENS.iconBox.borderRadius,  // 4px
}}
```
